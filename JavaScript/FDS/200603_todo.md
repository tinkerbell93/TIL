const render = () => {    
    let html = '';
    todos.forEach(todo => (
        // 문자열을 만든다.
        html += `
        <li id="${todo.id}" class="todo-item">
            <input id="ck-${todo.id}" class="checkbox" type="checkbox" ${todo.complated ? 'checked' : ''}>
            <label for="ck-${todo.id}">${todo.content}</label>
            <i class="remove-todo far fa-times-circle"></i>
        </li>`
    ));

};
